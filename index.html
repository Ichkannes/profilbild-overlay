<body>

    <h1>Profilbild freistellen mit Hintergrund und Overlay</h1>

    <canvas id="canvas"></canvas> <!-- Keine feste Breite/Höhe hier -->

    <br>

    <input type="file" id="fileInput" />

    <button id="downloadBtn">Bild herunterladen</button>


    <script>

        const canvas = document.getElementById('canvas');

        const ctx = canvas.getContext('2d');

        const fileInput = document.getElementById('fileInput');

        const downloadBtn = document.getElementById('downloadBtn');


        fileInput.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (!file) return;


            const img = new Image();

            img.onload = () => {

                // Dynamisch Canvas-Größe anpassen

                canvas.width = img.width;

                canvas.height = img.height;


                // Zeichne das hochgeladene Bild auf das Canvas

                ctx.drawImage(img, 0, 0, img.width, img.height);


                // Overlay hinzufügen

                const overlay = new Image();

                overlay.src = 'https://raw.githubusercontent.com/Ichkannes/profilbild-overlay/main/overlay.png';

                overlay.onload = () => {

                    // Überprüfen, ob das Overlay geladen wurde

                    ctx.drawImage(overlay, 0, 0, img.width, img.height);

                };

                overlay.onerror = () => {

                    alert("Fehler beim Laden des Overlays!");

                };

            };

            img.onerror = () => {

                alert("Fehler beim Laden des Bildes!");

            };

            img.src = URL.createObjectURL(file);

        });


        downloadBtn.addEventListener('click', () => {

            const link = document.createElement('a');

            link.download = 'edited-image.png';

            link.href = canvas.toDataURL();

            link.click();

        });

    </script>
